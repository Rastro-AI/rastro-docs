## Get Your API Key

<Steps>
  <Step title="Sign in to Dashboard">
    Go to [dashboard.rastro.ai](https://dashboard.rastro.ai) and sign in to your account.
  </Step>
  <Step title="Navigate to API Keys">
    Click **Settings** â†’ **API Keys** in the sidebar.
  </Step>
  <Step title="Create a New Key">
    Click **Create New Key**, give it a descriptive name (e.g., "Production API"), and click **Create**.
  </Step>
  <Step title="Copy Your Key">
    <Warning>
      Copy your API key immediately! For security, it will only be shown once.
    </Warning>
    Your key will look like: `rastro_pk_live_abc123...`
  </Step>
</Steps>

## Make Your First Request

Let's verify your API key works by listing your catalogs:

<CodeGroup>

```bash cURL
curl https://catalogapi.rastro.ai/api/public/catalogs \
  -H "Authorization: Bearer rastro_pk_your_key_here"
```

```python Python
import requests

response = requests.get(
    "https://catalogapi.rastro.ai/api/public/catalogs",
    headers={"Authorization": "Bearer rastro_pk_your_key_here"}
)

print(response.json())
```

```javascript JavaScript
const response = await fetch('https://catalogapi.rastro.ai/api/public/catalogs', {
  headers: {
    'Authorization': 'Bearer rastro_pk_your_key_here'
  }
});

const data = await response.json();
console.log(data);
```

```go Go
package main

import (
    "fmt"
    "net/http"
    "io"
)

func main() {
    client := &http.Client{}
    req, _ := http.NewRequest("GET", "https://catalogapi.rastro.ai/api/public/catalogs", nil)
    req.Header.Add("Authorization", "Bearer rastro_pk_your_key_here")

    resp, _ := client.Do(req)
    body, _ := io.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```

</CodeGroup>

## Create Your First Catalog

If you don't have any catalogs yet, create one:

<CodeGroup>

```bash cURL
curl -X POST https://catalogapi.rastro.ai/api/public/catalogs \
  -H "Authorization: Bearer rastro_pk_your_key_here" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "My Product Catalog",
    "unique_id_field": "sku",
    "description": "Products synced via API"
  }'
```

```python Python
import requests

response = requests.post(
    "https://catalogapi.rastro.ai/api/public/catalogs",
    headers={
        "Authorization": "Bearer rastro_pk_your_key_here",
        "Content-Type": "application/json"
    },
    json={
        "name": "My Product Catalog",
        "unique_id_field": "sku",
        "description": "Products synced via API"
    }
)

catalog = response.json()
print(f"Created catalog: {catalog['id']}")
```

```javascript JavaScript
const response = await fetch('https://catalogapi.rastro.ai/api/public/catalogs', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer rastro_pk_your_key_here',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    name: 'My Product Catalog',
    unique_id_field: 'sku',
    description: 'Products synced via API'
  })
});

const catalog = await response.json();
console.log(`Created catalog: ${catalog.id}`);
```

</CodeGroup>

## Add Products

Now let's add some products using the bulk upsert endpoint:

<CodeGroup>

```bash cURL
curl -X POST https://catalogapi.rastro.ai/api/public/catalogs/YOUR_CATALOG_ID/items/bulk \
  -H "Authorization: Bearer rastro_pk_your_key_here" \
  -H "Content-Type: application/json" \
  -d '{
    "items": [
      {
        "sku": "LAPTOP-001",
        "title": "Professional Laptop",
        "description": "High-performance laptop for professionals",
        "price": 1299.99,
        "category": "Electronics"
      },
      {
        "sku": "MOUSE-001",
        "title": "Wireless Mouse",
        "description": "Ergonomic wireless mouse",
        "price": 49.99,
        "category": "Electronics"
      }
    ],
    "unique_field": "sku"
  }'
```

```python Python
import requests

catalog_id = "YOUR_CATALOG_ID"

response = requests.post(
    f"https://catalogapi.rastro.ai/api/public/catalogs/{catalog_id}/items/bulk",
    headers={
        "Authorization": "Bearer rastro_pk_your_key_here",
        "Content-Type": "application/json"
    },
    json={
        "items": [
            {
                "sku": "LAPTOP-001",
                "title": "Professional Laptop",
                "description": "High-performance laptop",
                "price": 1299.99,
                "category": "Electronics"
            },
            {
                "sku": "MOUSE-001",
                "title": "Wireless Mouse",
                "description": "Ergonomic wireless mouse",
                "price": 49.99,
                "category": "Electronics"
            }
        ],
        "unique_field": "sku"
    }
)

result = response.json()
print(f"Created: {result['created']}, Updated: {result['updated']}")
```

## Run an AI Workflow

Execute an AI workflow to enrich your products:

<CodeGroup>

```bash cURL
# Start the workflow
curl -X POST https://catalogapi.rastro.ai/api/public/workflows/YOUR_WORKFLOW_ID/execute \
  -H "Authorization: Bearer rastro_pk_your_key_here" \
  -H "Content-Type: application/json" \
  -d '{
    "input": [
      {"sku": "LAPTOP-001", "title": "Professional Laptop"}
    ]
  }'
```

```python Python
import requests
import time

workflow_id = "YOUR_WORKFLOW_ID"
headers = {
    "Authorization": "Bearer rastro_pk_your_key_here",
    "Content-Type": "application/json"
}

# Start workflow
response = requests.post(
    f"https://catalogapi.rastro.ai/api/public/workflows/{workflow_id}/execute",
    headers=headers,
    json={
        "input": [
            {"sku": "LAPTOP-001", "title": "Professional Laptop"}
        ]
    }
)

run = response.json()
run_id = run["workflow_run_id"]
print(f"Started workflow: {run_id}")

# Poll for completion
while True:
    status = requests.get(
        f"https://catalogapi.rastro.ai/api/public/workflows/runs/{run_id}",
        headers=headers
    ).json()

    print(f"Status: {status['status']} ({status['progress']*100:.0f}%)")

    if status["status"] in ["completed", "failed"]:
        break

    time.sleep(3)

# Get results
if status["status"] == "completed":
    print("Results:", status["results"]["data"])
```

</CodeGroup>
