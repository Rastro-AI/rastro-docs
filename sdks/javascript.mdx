## Installation

```bash
npm install @rastro/sdk
# or
yarn add @rastro/sdk
```

## Quick Start

```typescript
import { Rastro } from '@rastro/sdk';

const client = new Rastro({ apiKey: 'rastro_pk_...' });

// List catalogs
const catalogs = await client.catalogs.list();

// Get items
const items = await client.catalogs.items.list('cat_abc123', { limit: 100 });

// Bulk upsert
const result = await client.catalogs.items.bulkUpsert('cat_abc123', {
  items: [
    { sku: 'A001', title: 'Product A', price: 29.99 },
    { sku: 'A002', title: 'Product B', price: 39.99 },
  ],
  uniqueField: 'sku'
});

console.log(`Created: ${result.created}, Updated: ${result.updated}`);
```

## Workflows

```typescript
// Execute and wait for results
const results = await client.workflows.executeAndWait('wf_abc123', {
  input: [
    { sku: 'A001', title: 'Product A' }
  ]
});

results.forEach(item => console.log(item));
```

## TypeScript Types

```typescript
import { Catalog, CatalogItem, WorkflowRun } from '@rastro/sdk';

const catalog: Catalog = await client.catalogs.get('cat_abc123');
const items: CatalogItem[] = await client.catalogs.items.list('cat_abc123');
```

## Error Handling

```typescript
import { Rastro, RastroError, AuthenticationError } from '@rastro/sdk';

try {
  const catalog = await client.catalogs.get('invalid_id');
} catch (error) {
  if (error instanceof AuthenticationError) {
    console.log('Invalid API key');
  } else if (error instanceof RastroError) {
    console.log(`API error: ${error.message}`);
  }
}
```
