## Installation

```bash
go get github.com/rastro-ai/rastro-go
```

## Quick Start

```go
package main

import (
    "fmt"
    "github.com/rastro-ai/rastro-go"
)

func main() {
    client := rastro.NewClient("rastro_pk_...")

    // List catalogs
    catalogs, _ := client.Catalogs.List(nil)
    fmt.Println(catalogs)

    // Get items
    items, _ := client.Catalogs.Items.List("cat_abc123", &rastro.ListOptions{
        Limit: 100,
    })

    // Bulk upsert
    result, _ := client.Catalogs.Items.BulkUpsert("cat_abc123", &rastro.BulkUpsertRequest{
        Items: []map[string]interface{}{
            {"sku": "A001", "title": "Product A", "price": 29.99},
            {"sku": "A002", "title": "Product B", "price": 39.99},
        },
        UniqueField: "sku",
    })

    fmt.Printf("Created: %d, Updated: %d\n", result.Created, result.Updated)
}
```

## Workflows

```go
// Execute and wait
results, err := client.Workflows.ExecuteAndWait("wf_abc123", []map[string]interface{}{
    {"sku": "A001", "title": "Product A"},
})

if err != nil {
    log.Fatal(err)
}

for _, item := range results {
    fmt.Println(item)
}
```

## Error Handling

```go
catalog, err := client.Catalogs.Get("invalid_id")
if err != nil {
    if rastro.IsAuthError(err) {
        fmt.Println("Invalid API key")
    } else if rastro.IsNotFoundError(err) {
        fmt.Println("Catalog not found")
    } else {
        fmt.Printf("Error: %v\n", err)
    }
}
```
